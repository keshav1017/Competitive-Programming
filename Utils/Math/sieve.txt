const int N = 1e7;
bool isPrime[N+1];
v(int) primes;
 
void sieveOfEratosthenes(){
    memset(isPrime, true, N+1);
    isPrime[0] = isPrime[1] = false;
    for(int i = 2; i*i <= N; i++){
        if(isPrime[i]){
            for(int j = i*i; j <= N; j += i){
                isPrime[j] = false;
            }
        }
    }
 
    for(int i = 2; i <= N; i++) if(isPrime[i]) primes.pb(i);
}
